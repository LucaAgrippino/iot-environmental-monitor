// cppcheck suppressions for the IoT Environmental Monitoring Gateway.
//
// Format: <error-id>:<file>:<line>   (file and line optional)
// Lines starting with '//' are comments.
//
// Each suppression must be accompanied by a comment explaining:
//   - which BARR-C rule or project constraint justifies the suppression, and
//   - why the flag is a false positive or an accepted deviation.

// missingInclude: vendored CMSIS not on cppcheck's include path; not a real defect.
// (CMSIS *is* reachable by the actual compiler — see firmware/<board>/.cproject.)
missingInclude

// unusedFunction: Phase 4 drivers are isolated until middleware/application consumers
// land (Logger, ModbusSlave, etc.). Cppcheck has no cross-module visibility from the
// CI invocation. Revisit at Phase 5 integration.
unusedFunction

// unusedStructMember: vtable function pointers in iXxx_t structs and
// pin descriptor fields in led_pin_t are consumed by downstream
// components not yet compiled in this CI invocation.
// Same justification as unusedFunction above.
unusedStructMember

// knownConditionTrueFalse:console_service.c — board_comms_ok() returns a constant
// in the unconfigured-board #else branch (neither BOARD_FIELD_DEVICE nor
// BOARD_GATEWAY defined). That branch is only reachable in non-production builds
// lacking a board macro. Production builds always define one of the two, so the
// branch is never compiled in the real firmware.
knownConditionTrueFalse:console_service.c

// redundantAssignment:lcd_driver.c — s_lcd_init_stage is a volatile debug
// sentinel written progressively through lcd_init() to mark each stage for
// observation via a hardware debugger live-watch. Writing without reading back
// in code is intentional; the volatile qualifier already prevents the compiler
// from eliding the writes. This is not a logic error.
redundantAssignment:lcd_driver.c

// staticFunction:cpu.c — cpu_delay_us and cpu_panic are public API declared in
// cpu.h. All in-TU callers happen to also live in cpu.c (cpu_delay_ms → delay_us;
// FreeRTOS hooks → panic), so cppcheck cannot see external consumers from a
// single-TU analysis. External linkage is required and intentional.
staticFunction:cpu.c

// duplicateExpression:cpu.c — cpu_delay_us uses DWT->CYCCNT, a volatile register
// that changes between reads. cppcheck incorrectly evaluates the loop subtraction
// (DWT->CYCCNT - start) as same-expression because start was captured from the
// same volatile address immediately before the loop.
duplicateExpression:cpu.c

// constParameterPointer:cpu.c — vApplicationStackOverflowHook follows the FreeRTOS
// weak-symbol ABI which specifies char* (non-const) for pcTaskName. Changing to
// const char* would break the ABI contract with the RTOS.
constParameterPointer:cpu.c
